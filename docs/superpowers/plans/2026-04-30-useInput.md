# useInput Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `useInput` hook to `@pilates/react` so applications can respond to keystrokes.

**Architecture:** Pure key-parser state machine + a refcounted `StdinProvider` with lazy raw-mode lifecycle, parallel to the existing `StdoutProvider`. `useInput` is a thin `useContext + useEffect` wrapper; subscribers are broadcast every keypress in mount order.

**Tech Stack:** TypeScript, `react@19`, `react-reconciler@0.31`, vitest. Lives entirely in `packages/react/` — no new top-level packages, no changes to `host-config.ts`.

**Spec:** `docs/superpowers/specs/2026-04-30-useInput-design.md`

---

## Conventions reused from v0.1

- `useApp` / `useStdout` / `useStderr` pattern in `hooks.ts`: `createContext<X | null>(null)`; hook calls `useContext` and throws "Pilates: useX() must be used inside <render>." if null. Match this exactly for `useInput`.
- `StdoutProvider` pattern in `render.tsx`: a wrapper component that owns the context value and lifecycle effects. `StdinProvider` mirrors this shape.
- `mount<T>` in `test-utils.ts`: returns a setState handle plus ANSI-write capture. `mountWithInput` extends this with `press*` helpers.
- `updateContainerSync` + `flushSyncWork` for synchronous commits in tests. Already wired in `test-utils.ts`; the keypress dispatch path triggers React updates that drain through this same mechanism.
- Commit message style: `feat(react): …`, `test(react): …`, `fix(react): …`, etc. One commit per task.

```ts
import { useEffect } from 'react';

export function useInput(
  handler: (event: KeyEvent) => void,
  options: UseInputOptions = {},
): void {
  const v = useContext(StdinContext);
  if (!v) throw new Error('Pilates: useInput() must be used inside <render>.');
  const isActive = options.isActive ?? true;
  useEffect(() => {
    const unsubscribe = v.subscribe(handler);
    if (!isActive) v.setActive(handler, false);
    return () => {
      unsubscribe();
    };
  }, [v, handler, isActive]);
}
```

`packages/react/src/key-parser.ts`:

```ts
import type { KeyEvent } from './hooks.js';

export interface ParseResult {
  events: KeyEvent[];
  remainder: string;
}

const ASCII_PRINTABLE_MIN = 0x20; // space
const ASCII_PRINTABLE_MAX = 0x7e; // ~

function isAsciiUppercase(ch: string): boolean {
  return ch.length === 1 && ch >= 'A' && ch <= 'Z';
}

export function parse(input: string): ParseResult {
  const events: KeyEvent[] = [];
  let i = 0;
  while (i < input.length) {
    const cp = input.codePointAt(i);
    if (cp === undefined) break;
    const ch = String.fromCodePoint(cp);
    const advance = ch.length;

    if (cp >= ASCII_PRINTABLE_MIN && cp <= ASCII_PRINTABLE_MAX) {
      events.push({
        ch,
        ctrl: false,
        alt: false,
        shift: isAsciiUppercase(ch),
        sequence: ch,
      });
      i += advance;
      continue;
    }

    // Non-printable byte we don't yet handle — emit raw.
    events.push({
      ctrl: false,
      alt: false,
      shift: false,
      sequence: ch,
    });
    i += advance;
  }
  return { events, remainder: '' };
}
```

```ts
    // ESC [ ... — CSI sequence. We need to scan for the terminator,
    // which is a final byte in the range 0x40..0x7e. If we run out of
    // input before finding one, this is a partial sequence; bail out
    // and report the remainder so the caller can buffer it.
    if (cp === 0x1b && input[i + 1] === '[') {
      const csiStart = i + 2; // first param byte (or final byte)
      let csiEnd = csiStart;
      while (csiEnd < input.length) {
        const code = input.charCodeAt(csiEnd);
        if (code >= 0x40 && code <= 0x7e) break; // final byte
        csiEnd++;
      }
      if (csiEnd >= input.length) {
        // Partial: stop parsing here; everything from i onward is remainder.
        return { events, remainder: input.slice(i) };
      }
      const params = input.slice(csiStart, csiEnd);
      const final = input[csiEnd]!;
      const sequence = input.slice(i, csiEnd + 1);
      const ev = decodeCsi(params, final, sequence);
      if (ev) events.push(ev);
      else events.push({ ctrl: false, alt: false, shift: false, sequence });
      i = csiEnd + 1;
      continue;
    }
```

Then add the `decodeCsi` helper at the bottom of the file:

```ts
function decodeCsi(params: string, final: string, sequence: string): KeyEvent | null {
  // Default modifiers; CSI modifier params (e.g., "1;5") are decoded later.
  const base: Pick<KeyEvent, 'ctrl' | 'alt' | 'shift' | 'sequence'> = {
    ctrl: false,
    alt: false,
    shift: false,
    sequence,
  };

  // Letter-final navigation: A/B/C/D, H, F.
  if (final === 'A') return { ...base, name: 'up' };
  if (final === 'B') return { ...base, name: 'down' };
  if (final === 'C') return { ...base, name: 'right' };
  if (final === 'D') return { ...base, name: 'left' };
  if (final === 'H') return { ...base, name: 'home' };
  if (final === 'F') return { ...base, name: 'end' };

  // Tilde-final: numeric param maps to a specific key.
  if (final === '~') {
    const num = parseInt(params, 10);
    switch (num) {
      case 3: return { ...base, name: 'delete' };
      case 5: return { ...base, name: 'pageUp' };
      case 6: return { ...base, name: 'pageDown' };
      case 15: return { ...base, name: 'f5' };
      case 17: return { ...base, name: 'f6' };
      case 18: return { ...base, name: 'f7' };
      case 19: return { ...base, name: 'f8' };
      case 20: return { ...base, name: 'f9' };
      case 21: return { ...base, name: 'f10' };
      case 23: return { ...base, name: 'f11' };
      case 24: return { ...base, name: 'f12' };
      default: return null;
    }
  }

  return null;
}
```

```ts
  // Modifier param decoding: xterm CSI uses `1;<mod>` for letter finals
  // and `<base>;<mod>` for tilde finals. <mod> = (shift|alt<<1|ctrl<<2) + 1.
  const parts = params.split(';');
  const baseStr = parts[0] ?? '';
  const modStr = parts[1] ?? '';
  let ctrl = false;
  let alt = false;
  let shift = false;
  if (modStr) {
    const mod = parseInt(modStr, 10) - 1;
    if (!Number.isNaN(mod)) {
      shift = (mod & 0b001) !== 0;
      alt = (mod & 0b010) !== 0;
      ctrl = (mod & 0b100) !== 0;
    }
  }
  const base = { ctrl, alt, shift, sequence };

  // Letter-final navigation: A/B/C/D, H, F.
  if (final === 'A') return { ...base, name: 'up' };
  if (final === 'B') return { ...base, name: 'down' };
  if (final === 'C') return { ...base, name: 'right' };
  if (final === 'D') return { ...base, name: 'left' };
  if (final === 'H') return { ...base, name: 'home' };
  if (final === 'F') return { ...base, name: 'end' };

  // Tilde-final: numeric base param maps to a specific key.
  if (final === '~') {
    const num = parseInt(baseStr, 10);
    switch (num) {
      case 3: return { ...base, name: 'delete' };
      case 5: return { ...base, name: 'pageUp' };
      case 6: return { ...base, name: 'pageDown' };
      case 15: return { ...base, name: 'f5' };
      case 17: return { ...base, name: 'f6' };

      case 19: return { ...base, name: 'f8' };
      case 20: return { ...base, name: 'f9' };
      case 21: return { ...base, name: 'f10' };
```

```tsx
import { parse as parseKeys } from './key-parser.js';
import {
  StdinContext,
  type StdinHookValue,
  type KeyEvent,
} from './hooks.js';

function StdinProvider({
  stdin,
  children,
}: {
  stdin: NodeJS.ReadStream;
  children?: ReactNode;
}) {
  // Subscribers and active-flags. Mutated imperatively across React commits;
  // raw mode is owned at the module/component level, not driven by render.
  const stateRef = useRef({
    subscribers: new Map<(event: KeyEvent) => void, boolean>(),
    refcount: 0,
    remainder: '',
    rawModeOn: false,
    listenerAttached: false,
  });
  const isRawModeSupported = stdin.isTTY === true;

  useEffect(() => {
    const state = stateRef.current;
    const onData = (chunk: Buffer | string) => {
      const text =
        typeof chunk === 'string'
          ? chunk
          : chunk.toString('utf8');
      const combined = state.remainder + text;
      const { events, remainder } = parseKeys(combined);
      state.remainder = remainder;
      for (const event of events) {
        for (const [handler, active] of state.subscribers) {
          if (!active) continue;
          try {
            handler(event);
          } catch (err) {
            const msg =
              err instanceof Error ? err.message : String(err);
            // Log but don't break other subscribers.
            process.stderr.write(
              `Pilates: useInput handler threw: ${msg}\n`,
            );
          }
        }
      }
    };
    // We attach onData up-front but only enter raw mode when refcount > 0.
    // Refcount changes happen in subscribe/setActive, called from React effects.
    return () => {
      stdin.off('data', onData);
      if (state.rawModeOn && isRawModeSupported) {
        try {
          stdin.setRawMode(false);
        } catch {
          /* swallow on teardown */
        }
      }
    };
  // We deliberately set up the listener once; subscribe/unsubscribe is dynamic.
  // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [stdin]);

  const value: StdinHookValue = {
    stdin,
    isRawModeSupported,
    subscribe: (handler) => {
      const state = stateRef.current;
      state.subscribers.set(handler, true);
      state.refcount += 1;
      ensureRawMode(stdin, state, isRawModeSupported);
      return () => {
        const wasActive = state.subscribers.get(handler) === true;
        state.subscribers.delete(handler);
        if (wasActive) {
          state.refcount -= 1;
          if (state.refcount === 0) releaseRawMode(stdin, state, isRawModeSupported);
        }
      };
    },
    setActive: (handler, active) => {
      const state = stateRef.current;
      const current = state.subscribers.get(handler);
      if (current === undefined) return;
      if (current === active) return;
      state.subscribers.set(handler, active);
      if (active) {
        state.refcount += 1;
        ensureRawMode(stdin, state, isRawModeSupported);
      } else {
        state.refcount -= 1;
        if (state.refcount === 0) releaseRawMode(stdin, state, isRawModeSupported);
      }
    },
  };
  return createElement(StdinContext.Provider, { value }, children);
}

function ensureRawMode(
  stdin: NodeJS.ReadStream,
  state: { rawModeOn: boolean; listenerAttached: boolean; remainder: string; subscribers: Map<unknown, unknown>; refcount: number },
  isRawModeSupported: boolean,
): void {
  if (!state.listenerAttached) {
    state.listenerAttached = true;
    // The 'data' handler is bound by the useEffect closure above; nothing
    // to attach here. We keep the flag to make tests legible.
  }
  if (!state.rawModeOn && isRawModeSupported) {
    try {
      stdin.setRawMode(true);
      stdin.resume();
      state.rawModeOn = true;
    } catch (err) {
      const msg = err instanceof Error ? err.message : String(err);
      process.stderr.write(
        `Pilates: raw stdin mode not available; keystroke dispatch may be line-buffered (${msg})\n`,
      );
    }
  } else if (!isRawModeSupported) {
    // Non-TTY: just resume so 'data' fires. setRawMode is unsupported.
    try { stdin.resume(); } catch { /* ignore */ }
  }
}

function releaseRawMode(
  stdin: NodeJS.ReadStream,
  state: { rawModeOn: boolean },
  isRawModeSupported: boolean,
): void {
  if (state.rawModeOn && isRawModeSupported) {
    try {
      stdin.setRawMode(false);
      state.rawModeOn = false;
    } catch {
      /* swallow */
    }
  }
}
```
